let xp = parseInt(localStorage.getItem("xp")) || 0; let level = parseInt(localStorage.getItem("level")) || 1; function update(){ document.getElementById("xp").innerText = xp; document.getElementById("lvl").innerText = level; } update(); /* 🔓 UNLOCK SYSTEM */ function unlocks(){ let snake = document.getElementById("snake"); let dodge = document.getElementById("dodge"); if(level >= 2){ snake.onclick = () => openGame("snake"); } else { snake.classList.add("locked"); snake.innerText = "🐍 Snake (Lvl 2)"; } if(level >= 3){ dodge.onclick = () => openGame("dodge"); } else { dodge.classList.add("locked"); dodge.innerText = "🏃 Dodge (Lvl 3)"; } } unlocks(); /* 🎮 OPEN GAME */ function openGame(game){ localStorage.setItem("nextGame", game); window.location.href = "loading.html"; } /* 🧠 KEY EXIT */ document.addEventListener("keydown",(e)=>{ if(e.key.toLowerCase()=="m"){ window.location.href="arcade.html"; } });
